```bash
#!/bin/bash
# 网络连通性检测脚本示例

# 定义目标主机列表
HOSTS=("192.168.1.1" "8.8.8.8" "10.0.0.1" "gateway")

# 网络测试函数
function test_network() {
    local host=$1
    ping -c 3 -W 2 $host > /dev/null 2>&1
    
    if [ $? -eq 0 ]; then
        echo "$host 连接正常"
    else
        echo "$host 无法连接！"
    fi
}

# 主函数
echo "开始网络测试..."
for host in "${HOSTS[@]}"; do
    test_network $host
done
echo "网络测试完成。"
``` 
